# Question 1 Part (a)
true_a = 2.0
true_b = 1.0
sigma_noise = 0.5 # The Question says its centered on 0

def linear_model(x, a, b):
    return a + b * x

def generate_and_fit(N=11, a=true_a, b=true_b, sigma=sigma_noise, return_data=False):
    # Setup numpy default rng (As seen in Analysis Assignment 1)
    rng = np.random.default_rng() # Random Generator function
    
    x = np.arange(N) # Number 0 to 10
    
    noise = rng.normal(0.0, sigma, N) # Normal or Gaussian Noise
    
    y = a + b * x + noise
    
    popt, pcov = curve_fit(linear_model, x, y, sigma=np.full(N, sigma), absolute_sigma=True)
    
    a_est, b_est = popt # Estimates
    a_err, b_err = np.sqrt(np.diag(pcov))
    covariance = pcov[0, 1]
    correlation = covariance / (a_err * b_err)

    if return_data: # Just an Argument I made so that I can opt to return the Random Data points as well
        return x, y, a_est, b_est, a_err, b_err, correlation
    
    return a_est, b_est, a_err, b_err, correlation

x_sample, y_sample, a_est, b_est, a_err, b_err, corr = generate_and_fit(return_data=True)

print(f"Results from a single trial (N=11):")
print(f"Estimated intercept a: {a_est:.4f} +/- {a_err:.4f}")
print(f"Estimated slope b: {b_est:.4f} +/- {b_err:.4f}")
print(f"Correlation coefficient: {corr:.4f}")

plt.figure(figsize=(8, 5))
plt.errorbar(x_sample, y_sample, yerr=sigma_noise, fmt='o', label='Simulated Data')
plt.plot(x_sample, linear_model(x_sample, true_a, true_b), 'k--', label='True Line')
plt.plot(x_sample, linear_model(x_sample, a_est, b_est), 'r-', label='Best Fit')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.show()

Results from a single trial (N=11):
Estimated intercept a: 1.7974 +/- 0.2820
Estimated slope b: 1.0388 +/- 0.0477
Correlation coefficient: -0.8452

# Question 2 Part (c)
N_boot = 1000
a_boot = np.zeros(N_boot)
b_boot = np.zeros(N_boot)
a_cum_mean = np.zeros(N_boot)
b_cum_mean = np.zeros(N_boot)
a_cum_std = np.zeros(N_boot)
b_cum_std = np.zeros(N_boot)

for i in range(N_boot): # Interate
    # Using rng.choice to sample with replacement, just found this
    indices = rng.choice(N_data2, N_data2, replace=True)
    x_syn = x_data[indices]
    y_syn = y_data[indices]
    
    # Here we fit the line to the Self geenrated Data
    popt_syn, _ = curve_fit(linear_model, x_syn, y_syn)
    a_boot[i] = popt_syn[0] # Storing it respectively 
    b_boot[i] = popt_syn[1]
    
    a_cum_mean[i] = np.mean(a_boot[:i+1])
    b_cum_mean[i] = np.mean(b_boot[:i+1])
    if i == 0:
        a_cum_std[i] = 0.0
        b_cum_std[i] = 0.0
    else:
        a_cum_std[i] = np.std(a_boot[:i+1], ddof=1)
        b_cum_std[i] = np.std(b_boot[:i+1], ddof=1)
